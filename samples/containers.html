Dino.widget({
  dtype: 'box',
  items: [{
    dtype: 'tabs',
    defaultItem: {
      baseCls: 'tab',
      cls: 'ui-widget-header ui-corner-top'
    },
    tabs: [
      { text: 'Tab1' }, 
      {	text: 'Tab2' }
    ],
    onTabChanged: function() {
      this.parent.getItem('stack').layout.activate(this.currentTab.index);
    }
  }, {
    tag: 'stack',
    layout: 'stack-layout',
    items: [{
      text: 'Содержимое контейнера №1'
    }, {
      text: 'Содержимое контейнера №2'
    }]
  }]
});
