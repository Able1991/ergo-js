Переключение состояний это просто добавление/удаление классов, имеющих специальные имена. Метод toggleState в действительности вызывает метод jQuery.toggleClass и является лишь формой упрощенной записи. 

new Dino.Widget('<div/>', {
  baseCls: 'colored-box',
  text: 'Кликни по мне',
  width: 100,
  height: 100,
  events: {
    'click': function() { $(this).dino().toggleState('click'); }
  },
  onStateChanged: function() {
    growl.show('Состояние изменилось');
  }
});

Dino использует механизм событий jQuery и свою собственную релизацию диспетчера событий в классе Dino.events.Observer. Обработчики событий jQuery определяются в блоке events, обработчики внутренних событий начинаются с on и заглавной буквы, например onLoad. 

new Dino.Widget('<div/>', {
  baseCls: 'colored-box',
  text: 'Наведи на меня курсор',
  width: 100,
  height: 100,
  events: {
    'mouseenter': function() { $(this).dino().events.fire('onCustomAction1'); },
    'mouseleave': function() { $(this).dino().events.fire('onCustomAction2'); }
  },
  onCustomAction1: function() {
    this.setState('hover');
  },
  onCustomAction2: function() {
    this.clearState('hover');
  }
});
